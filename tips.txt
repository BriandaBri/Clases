################# COMANDOS GIT #################

git clone << URL_REPOSITORIO >> -> clonar el repositorio en tu local
git checkout -b <<nombre_branch>> -> crear la rama
git checkout <<nombre_branch>> -> acceder a la rama
git commit -m <<"mensaje">> -> enviar mensaje de subida al repo
git add <<nombre de fichero> -> añadir fichero nuevo al repo
git push origin <<rama destino>> -> guardar los datos en el repo
